        src/RsaJsonRpcActivatorTestSuite.cc
        src/RsaJsonRpcTestSuite.cc
)
celix_deprecated_utils_headers(test_rsa_json_rpc)

    Celix::c_rsa_spi
    calculator_api
    Celix::rsa_common
    libuuid::libuuid
    jansson::jansson
    Celix::dfi
    Celix::framework
    Celix::shell_api
    GTest::gtest
    GTest::gtest_main
    )
